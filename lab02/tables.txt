№14. Списание оборудования. 


ОБОРУДОВАНИЕ
	* Название ( varchar(50) )
	Тип ( varchar(20) )


РАБОТНИКИ
	* Фамилия ( varchar(25) )
	Имя ( varchar(15) )
	Отчество ( varchar(20) )
	Должность ( varchar(50) )
	Дата_приема_на_работу ( date )


ПОСТУПЛЕНИЯ
	* Номер_поступления ( varchar(10) )
	Название_оборудования ( varchar(50) )
	Дата_поступления ( datetime )
	Количество ( int )
	Подразделение_установки ( varchar(50) )
	Фамилия_ответственного ( varchar(25) )


СПИСАНИЯ
	* Номер_списания ( varchar(10) )
	Номер_поступления ( varchar(10) )
	Количество ( int )
	Дата_списания ( datetime )
	Причина_списания ( varchar(100) )
	Фамилия_ответственного ( varchar(25) )
